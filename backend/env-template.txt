# Vercel Environment Variables Template
# Copy this content to your .env.local file
# Replace the placeholder values with your actual values

# Database Configuration
DATABASE_URL=postgresql://your-username:your-password@your-host:5432/your-database?sslmode=require

# JWT Configuration
JWT_SECRET=your-super-secret-jwt-key-min-32-characters-long

# Environment
NODE_ENV=production

# Instructions:
# 1. Get your Neon Database URL from https://neon.tech
# 2. Replace the DATABASE_URL above with your actual Neon URL
# 3. Generate a secure JWT_SECRET (use: openssl rand -base64 32)
# 4. Save this file as .env.local (replace the content)
# 5. Run: vercel env add DATABASE_URL (paste your Neon URL)
# 6. Run: vercel env add JWT_SECRET (paste your JWT secret)
# 7. Deploy: vercel --prod
